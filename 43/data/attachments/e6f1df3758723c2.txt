const response = pm.response.json();

pm.test("Data object should exist", function () {
    pm.expect(response.data).to.exist;
});

pm.test("Data object should have the required fields", function () {
    pm.expect(response.data).to.have.property('id');
    pm.expect(response.data).to.have.property('first_name');
    pm.expect(response.data).to.have.property('last_name');
    pm.expect(response.data).to.have.property('email');
    pm.expect(response.data).to.have.property('phone_number');
    pm.expect(response.data).to.have.property('status');
    pm.expect(response.data).to.have.property('last_login_at');
});

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response message should indicate successful retrieval", function () {
    pm.expect(response.message).to.equal("Customer retrieved successfully");
});
