const jsonData = pm.response.json();

pm.test("Status code is 201", function () {
    pm.response.to.have.status(201);
});

pm.test("Validate the response data", function () {
    pm.expect(jsonData.status).to.exist;
    pm.expect(jsonData.message).to.exist;
    pm.expect(jsonData.data).to.be.an("object");
    pm.expect(jsonData.data.meta).to.be.an("object");
    pm.expect(jsonData.data.account_name).to.exist;
    pm.expect(jsonData.data.meta.bank_name).to.exist;
})
